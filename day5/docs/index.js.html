/**
 * 
 * @param {Number} nilai 
 */
function getGrade(nilai) {
  var grade;
  if(nilai > 80) {
    grade = "A"
  } else if(nilai > 60 && nilai < 80) {
    grade = "B"
  }
  return grade
}
